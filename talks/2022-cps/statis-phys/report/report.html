\begin{eqnarray} \mathcal{Z}&=&\frac{1}{N!h^{3N}}\int d\vec{\textbf{p}}^N\int d\vec{\textbf{q}}^N\exp\{-\beta[\sum_{i=1}^{N}\frac{\vec{\textbf{p}}_i^2}{2m}+U(\vec{\textbf{q}}^N)]\} \\ &=&\frac{1}{N!}\left(\frac{2\pi m}{\beta h^2}\right)^{\frac{3}{2}N}\int d\vec{\textbf{q}}^N\exp[-\beta U(\vec{\textbf{q}}^N)] \end{eqnarray} \begin{equation} \mathcal{Q}=\int d\vec{\textbf{q}}^N\exp[-\beta U(\vec{\textbf{q}}^N)] \longrightarrow {\text{\color{red} \Large 位形积分} (3N\text{-dimensional})} \end{equation}

Direct Integral Approach (DIA):高维积分的重新诠释

$$I_{1D}\rightarrow\text{等效长度}$$

$$I_{2D}\rightarrow\text{等效面积}$$

\begin{equation} I_{ND}\longrightarrow N\text{维等效体积} \end{equation}

\begin{eqnarray} \mathcal Q&=&\int dq^{3N}\exp[-\beta U(q^{3N})], \\ \textcolor{red}{U'(q'^{3N})}&\textcolor{red}{=}&\textcolor{red}{U(q^{3N})-U_0},\quad \textcolor{red}{q'_i=q_i-q_0},\\ \mathcal{Q}&=&e^{-\beta U_0}\int dq'^{3N}\exp[-\beta U'(q'^{3N})]=e^{-\beta U_0}\mathcal{Q}'\\ \mathcal{Q}'&\simeq&\prod_{i=1}^{3N}\mathcal{L}_i, \quad\mathcal{L}_i=\int e^{-\beta U'(0\ldots q'_i\ldots 0)}dq'_i\\ \end{eqnarray} $$\Downarrow$$ \begin{eqnarray} &\mathcal{Q}=e^{-\beta U_0}\mathcal{L}^{3N}\\ &{\color{red}3N\text{重积分}\rightarrow \text{1重积分}} \end{eqnarray}

通过DIA缩小3N维等效体积

\begin{eqnarray} &&I_{2D}=\int_0^{a_1}dx\int_0^{a_2}dye^{-U(x,y)},\\ &&X'(x)=\int_0^xe^{-U(\xi,0)}d\xi, Y'(y)=\int_0^ye^{-U(0,\xi)}d\xi.\\ &&X'(0)=0, Y'(0)=0\\ &&a'_1=\int_0^{a_1}e^{-U(\xi,0)}d\xi,a'_2=\int_0^{a_2}e^{-U(0,\xi)}d\xi\\ &&I_{2D}=\int_0^{a'_1}dX'\int_0^{a'_2}dY'e^{-F(X',Y')}.\\ &&\text{If U(x,y) is positive definite and has a minimum at the origin (U(0,0)=0)},\\ &&F(0,0)=0,\frac{\partial F}{\partial X'}{\Big |}_0=\frac{\partial F}{\partial Y'}{\Big |}_0=0\\ &&I_{2D}\simeq a'_1a'_2. \end{eqnarray}
